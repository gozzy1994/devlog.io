// Building a rate limiter in Go
func NewLimiter(rate int) *Limiter {
  return &Limiter{
    tokens: make(chan struct{}, rate),
    rate: time.Second / time.Duration(rate),
  }
}

func (l *Limiter) Allow() bool {
  select {
  case l.tokens <- struct{}{}:
    return true
  default:
    return false
  }
}
